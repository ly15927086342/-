3. 防抖
```javascript
// 题目
const newFn = debounce(fn, 5000)

newFn(1,2,34,5,6)
newFn()
newFn()
newFn()
newFn()

// 解答
function debounce(fn,timeout){
	var st = null
	return function(){
		if(st){
			clearTimeout(st)
		}
		st = setTimeout(fn,timeout,arguments)
	}
}
```
